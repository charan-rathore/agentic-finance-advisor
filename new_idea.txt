class SourceRegistry(Base):
    """
    Tracks every external data source the system has fetched from.
    One row per unique URL fetched. Upserted on every fetch cycle.
    """
    __tablename__ = "source_registry"

    id = Column(Integer, primary_key=True, autoincrement=True)
    url = Column(String(1000), nullable=False, unique=True, index=True)
    domain = Column(String(200), nullable=False, index=True)
    source_name = Column(String(200))          # e.g. "SEC EDGAR", "Yahoo Finance RSS"
    source_type = Column(String(50))           # "sec_filing" | "rss_news" | "macro" | "reddit" | "price"
    is_trusted = Column(Boolean, default=False) # passes domain whitelist check
    is_reachable = Column(Boolean, default=True)
    http_status = Column(Integer)              # last HTTP status code seen
    first_fetched_at = Column(DateTime, default=datetime.utcnow)
    last_fetched_at = Column(DateTime, default=datetime.utcnow)
    fetch_count = Column(Integer, default=1)


class KnowledgeVersion(Base):
    """
    Audit trail for every wiki page update.
    One row per write to any data/wiki/*.md file.
    Answers: what changed, when, and what source drove the change.
    """
    __tablename__ = "knowledge_versions"

    id = Column(Integer, primary_key=True, autoincrement=True)
    page_name = Column(String(300), nullable=False, index=True)  # e.g. "stocks/AAPL.md"

    source_urls = Column(Text)                                    # JSON list of source URLs that drove this update

    word_count_after = Column(Integer, nullable=False)

# ── Source Validation ─────────────────────────────────────────────────────────

def extract_domain(url: str) -> str:
    """Extract the registered domain from a URL."""
    try:
        parsed = urlparse(url)
        return parsed.netloc.lower().lstrip("www.")
    except Exception:
        return ""


def is_trusted_domain(url: str) -> bool:
    """Check if the URL's domain is in the trusted whitelist."""
    domain = extract_domain(url)
    # Check exact match and parent domain match
    return domain in TRUSTED_DOMAINS or any(
        domain.endswith("." + trusted) or domain == trusted
        for trusted in TRUSTED_DOMAINS
    )


def validate_source(url: str, timeout: float = 5.0) -> dict:
    """
    Validate a source URL: check domain trust + HTTP reachability.

    Returns a dict with:
      is_trusted: bool       — passes domain whitelist
      is_reachable: bool     — HTTP request succeeded
      http_status: int|None  — status code returned
      domain: str            — extracted domain

    Uses HEAD request first (cheap), falls back to GET if HEAD returns 405.
    Never raises — all errors are caught and returned as is_reachable=False.
    """
    domain = extract_domain(url)
    trusted = is_trusted_domain(url)
    result = {
        "url": url,
        "domain": domain,
        "is_trusted": trusted,
        "is_reachable": False,
        "http_status": None,
    }

    try:
        with httpx.Client(timeout=timeout, follow_redirects=True) as client:
            try:
                r = client.head(url)
                result["http_status"] = r.status_code
                result["is_reachable"] = r.status_code < 400
            except Exception:
                # HEAD failed — try GET with limited content
                r = client.get(url, headers={"Range": "bytes=0-0"})
                result["http_status"] = r.status_code
                result["is_reachable"] = r.status_code < 400
    except Exception as e:
        logger.debug(f"[Trust] Reachability check failed for {url}: {e}")

    return result


# ── Source Registry ───────────────────────────────────────────────────────────

def register_source(
    engine,
    url: str,
    source_name: str,
    source_type: str,
    validation_result: dict | None = None,
) -> None:
    """
    Upsert a source into the source_registry table.

    If the URL already exists: increment fetch_count, update last_fetched_at
    and any new validation results.
    If new: create a full record including trust/reachability status.

    Call this from core/fetchers.py after every successful data fetch.
    """
    if validation_result is None:
        validation_result = {
            "domain": extract_domain(url),
            "is_trusted": is_trusted_domain(url),
            "is_reachable": True,
            "http_status": 200,
        }

    with Session(engine) as session:
        existing = session.query(SourceRegistry).filter_by(url=url).first()
        now = datetime.now(timezone.utc)

        if existing:
            existing.last_fetched_at = now
            existing.fetch_count = (existing.fetch_count or 0) + 1
            existing.is_reachable = validation_result.get("is_reachable", True)
            existing.http_status = validation_result.get("http_status")
            existing.is_trusted = validation_result.get("is_trusted", False)
        else:
            record = SourceRegistry(
                url=url,
                domain=validation_result.get("domain", extract_domain(url)),
                source_name=source_name,
                source_type=source_type,
                is_trusted=validation_result.get("is_trusted", False),
                is_reachable=validation_result.get("is_reachable", True),
                http_status=validation_result.get("http_status"),
                first_fetched_at=now,
                last_fetched_at=now,
                fetch_count=1,
            )
            session.add(record)

        session.commit()
        logger.debug(f"[Trust] Registered source: {url} (trusted={validation_result.get('is_trusted')})")


# ── Knowledge Versioning ──────────────────────────────────────────────────────

def record_wiki_version(
    engine,
    page_name: str,
    new_content: str,
    old_content: str,
    change_summary: str,
    source_urls: list[str],
    source_types: list[str],
    triggered_by: str = "ingest",
) -> int:
    """
    Record a new version entry whenever a wiki page is written.

    Computes the next version number for the page (max existing + 1).
    Stores word counts before/after so growth of the knowledge base
    is measurable over time.

    Returns the new version number.
    """
    with Session(engine) as session:
        last = (
            session.query(KnowledgeVersion)
            .filter_by(page_name=page_name)
            .order_by(KnowledgeVersion.version.desc())
            .first()
        )
        next_version = (last.version + 1) if last else 1

        entry = KnowledgeVersion(
            page_name=page_name,
            version=next_version,
            changed_at=datetime.now(timezone.utc),
            change_summary=change_summary,
            source_urls=json.dumps(source_urls),
            source_types=json.dumps(source_types),
            word_count_before=len(old_content.split()) if old_content else 0,
            word_count_after=len(new_content.split()),
            triggered_by=triggered_by,
        )
        session.add(entry)
        session.commit()
        logger.debug(f"[Trust] Versioned {page_name} → v{next_version} ({triggered_by})")
        return next_version


def get_page_version_history(engine, page_name: str) -> list[dict]:
    """
    Return the full version history for a wiki page, newest first.
    Used by the UI Sources & History page.
    """
    with Session(engine) as session:
        rows = (
            session.query(KnowledgeVersion)
            .filter_by(page_name=page_name)
            .order_by(KnowledgeVersion.version.desc())
            .all()
        )
        return [
            {
                "page": r.page_name,
                "version": r.version,
                "changed_at": r.changed_at,
                "change_summary": r.change_summary,
                "source_urls": json.loads(r.source_urls or "[]"),
                "source_types": json.loads(r.source_types or "[]"),
                "word_count_before": r.word_count_before,
                "word_count_after": r.word_count_after,
                "triggered_by": r.triggered_by,
            }
            for r in rows
        ]


def get_all_sources(engine) -> list[dict]:
    """
    Return all registered sources for the UI Sources & History page.
    Sorted by last_fetched_at descending.
    """
    with Session(engine) as session:
        rows = (
            session.query(SourceRegistry)
            .order_by(SourceRegistry.last_fetched_at.desc())
            .all()
        )
        return [
            {
                "url": r.url,
                "domain": r.domain,
                "source_name": r.source_name,
                "source_type": r.source_type,
                "is_trusted": r.is_trusted,
                "is_reachable": r.is_reachable,
                "http_status": r.http_status,
                "first_fetched_at": r.first_fetched_at,
                "last_fetched_at": r.last_fetched_at,
                "fetch_count": r.fetch_count,
            }
            for r in rows
        ]

def _compute_confidence(consulted_pages: list[str]) -> float:
    """
    Honest, explainable confidence score based on three observable signals.
    Documented in code so it can be explained clearly in interviews.
    """
    import yaml as _yaml
    from datetime import timezone as _tz

    score = 1.0
    source_types_seen = set()
    now = datetime.now(timezone.utc)

    for page in consulted_pages:
        content = _read_wiki_file(page)
        if not content:
            continue

        # Parse YAML frontmatter if present
        if content.startswith("---"):
            try:
                fm_text = content.split("---")[1]
                fm = _yaml.safe_load(fm_text) or {}
            except Exception:
                fm = {}
        else:
            fm = {}

        # Stale penalty
        if fm.get("stale") is True:
            score -= 0.15

        # Source type diversity
        for st in fm.get("data_sources", []):
            source_types_seen.add(st)

        # Recency penalty
        lu = fm.get("last_updated")
        if lu:
            try:
                if isinstance(lu, str):
                    lu_dt = datetime.fromisoformat(lu.replace("Z", "+00:00"))
                else:
                    lu_dt = lu.replace(tzinfo=timezone.utc) if lu.tzinfo is None else lu
                age_hours = (now - lu_dt).total_seconds() / 3600
                if age_hours > 24:
                    score -= 0.10
            except Exception:
                pass

    if len(source_types_seen) < 2:
        score -= 0.20

    return round(max(score, 0.30), 2)


def _any_stale(consulted_pages: list[str]) -> str:
    """Return 'yes' if any consulted page has stale:true in frontmatter, else 'no'."""
    import yaml as _yaml
    for page in consulted_pages:
        content = _read_wiki_file(page)
        if content and content.startswith("---"):
            try:
                fm = _yaml.safe_load(content.split("---")[1]) or {}
                if fm.get("stale") is True:
                    return "yes"
            except Exception:
                pass
    return "no"
